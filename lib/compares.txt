function number string:compare(BS:string){
    if (This == BS){
        return 0
    }
    
    for(I = 1, This:length()){
        if (BS:length() < I){
            return 1
        }
        
        local Abyte = toByte(This[I])
        local Bbyte = toByte(BS[I])
        
        if (Abyte == Bbyte){
        }elseif (Abyte < Bbyte){
            #print("Abyte:" + Abyte + " < Bbyte:" + Bbyte)
            return -1
        }else{
            #print("Abyte:" + Abyte + " < Bbyte:" + Bbyte)
            return 1
        }
    }
    return -1
}

function number number:compare(BS:number){
    if (This == BS){
        return 0
    }elseif (This < BS){
        return -1
    }else{
        return 1
    }
}
